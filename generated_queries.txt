SELECT 
    s.seller_id
FROM sellers s
JOIN order_items oi ON s.seller_id = oi.seller_id
JOIN orders o ON oi.order_id = o.order_id
JOIN customers c ON o.customer_id = c.customer_id
WHERE c.customer_city = 'rio de janeiro'
    AND o.order_status = 'delivered'
GROUP BY s.seller_id
ORDER BY COUNT(DISTINCT o.order_id) DESC
LIMIT 1;

SELECT ROUND(AVG(r.review_score), 2) AS score
FROM order_reviews r
JOIN orders o ON r.order_id = o.order_id
JOIN order_items oi ON o.order_id = oi.order_id
JOIN products p ON oi.product_id = p.product_id
WHERE o.order_status = 'delivered' AND p.product_category_name = 'beleza_saude';

SELECT COUNT(*) AS seller_count
FROM (
    SELECT s.seller_id
    FROM sellers s
    JOIN order_items oi ON s.seller_id = oi.seller_id
    JOIN orders o ON oi.order_id = o.order_id
    WHERE o.order_status = 'delivered'
    GROUP BY s.seller_id
    HAVING SUM(oi.price) > 100000
) high_value_sellers;

SELECT 
    p.product_category_name AS category_name
FROM products p
JOIN order_items oi ON p.product_id = oi.product_id
JOIN orders o ON oi.order_id = o.order_id
JOIN order_reviews r ON o.order_id = r.order_id
WHERE o.order_status = 'delivered'
GROUP BY p.product_category_name
ORDER BY 
    SUM(r.review_score = 5) * 1.0 / COUNT(r.review_score) DESC
LIMIT 1;

SELECT op.payment_installments AS installments
FROM orders o
JOIN order_payments op ON o.order_id = op.order_id
WHERE o.order_id IN (
    SELECT oi.order_id
    FROM order_items oi
    GROUP BY oi.order_id
    HAVING SUM(oi.price) + SUM(oi.freight_value) > 1000
)
GROUP BY op.payment_installments
ORDER BY COUNT(*) DESC
LIMIT 1;

Query Result:     installments
0            10

SELECT c.customer_city AS city_name
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
JOIN customers c ON o.customer_id = c.customer_id
GROUP BY c.customer_city
ORDER BY AVG(oi.freight_value) DESC
LIMIT 1;

SELECT 
    p.product_category_name AS category_name
FROM 
    products p
JOIN 
    order_items oi ON p.product_id = oi.product_id
JOIN 
    orders o ON oi.order_id = o.order_id
WHERE 
    o.order_status = 'delivered' 
    AND o.order_delivered_customer_date IS NOT NULL
GROUP BY 
    p.product_category_name
ORDER BY 
    AVG(JULIANDAY(o.order_delivered_customer_date) - JULIANDAY(o.order_purchase_timestamp)) ASC
LIMIT 1;

Query Result:          category_name
0  artes_e_artesanato
